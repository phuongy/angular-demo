Loading the data based on the view



    function UserCtrl(UserFactory, UserService) {
        this.users = UserFactory.users;

        this.updateList = function() {
            this.users = UserFactory.users;
        }

        this.resolve = function() {
            var self = this;

            UserService
                .getAll()
                .then(function(response) {
                    UserFactory.set(response.data);
                    self.updateList;
                });
        }
    }

    angular
        .module('app')
        .controller('UserCtrl', UserCtrl);
