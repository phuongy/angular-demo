Factory and Model objects


    var User() {

    }

    angular
        .module('app')
        .factory('UserFactory', function() {
        
            var UserFactory = {};
            UserFactory.users = [];

            UserFactory.add = function(User) {
                User.id = UserFactory.users.length;
                UserFactory.users.push(User);
            }

            UserFactory.get = function(id) {
                return UserFactory.users[id];
            }

            return UserFactory;
        });
