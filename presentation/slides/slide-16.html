Services and AJAX calls


    angular
        .module('app')
        .service('UserService', function( $http, $q ) {

            function add(user) {

                return $http.post('user/add', {
                    params: { user: user}
                }).then(function(response) {
                    return response.data;
                }, function(error) {
                    return "error";
                });
            
            }

            return {
                add: add
            }
        
        });
