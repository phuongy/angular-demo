Setting up the dynamic views


    angular
        .module('app')
        .config(['$routeProvider', function ($routeProvider) {

            $routeProvider
                .when('/users', {
                    templateUrl: 'views/users/main.html'
                })
                .when('/user/:userid', {
                    templateUrl: 'views/users/view.html'
                })
                .otherwise({
                    redirectTo: '/views/main.html'
                });
        }]);
