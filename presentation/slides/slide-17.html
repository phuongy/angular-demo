Dependency Injection


    angular
        .module('app')
        .controller('PageCtrl', function(UserFactory, UserService) {
    
            this.users = UserFactory.users;

            this.addUser = function(user) {
                UserService
                    .add(user)
                    .then(function() {
                        UserFactory.add(user);
                    });

            }

            this.updateList = function() {
                this.users = UserFactory.users;
            }

        });
